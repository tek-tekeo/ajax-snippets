<template>
  <span> 
    <v-btn v-if="btnColor != ''"
      x-large
      :color="btnColor"
      :href="url"
      class="ma-2 white--text"
    >
    {{ content }}
    <v-icon>
        mdi-arrow-right-bold-box
      </v-icon>
    </v-btn>
    <a v-else
      :href="url"
      @click="clickRecord"
      rel="nofollow noopener"
      target="_blank"
      v-html="redirectURL"
    >
    </a>
    <img
      v-if="imgTag != null"
      border="0"
      width="1"
      height="1"
      :src="imgTag"
      alt=""
    >
  </span>
</template>

<script>
module.exports = {
  data(){
    return{
    }
  },
  computed:{
    redirectURL(){
      if(this.setBanner == 'true'){
        return '<img border="0" width="'+this.imgWidth+'" height="'+this.imgHeight+'" alt="'+this.content+'" src="'+this.imgSrc+'">';
      }
      return this.content;
    }
  },
  methods: {
    clickRecord(){
      const log = {itemId:this.itemId, place:this.place};
      this.$emit('click-record', log);
    }
  },
  props: {
    imgSrc: {
      type: String,
      default: null
    },
    imgWidth: {
      type: Number,
      default: 300
    },
    imgHeight: {
      type: Number,
      default: 250
    },
    content: {
      type: String,
      default: '公式サイト'
    },
    url: {
      type: String,
      default: '#'
    },
    place: {
      type: String,
      default: 'none'
    },
    itemId: {
      type: Number,
      default: null
    },
    imgTag:{
      type: String,
      default: null
    },
    btnColor:{
      type: String,
      default: ''
    },
    setBanner:{
      type: String,
      default: 'false'
    }
  }
};
</script>
